	public ListNode reverse(ListNode list){
      if(list == null) return null;

	  ListNode next = list.next;
	  ListNode cur = list;
	  ListNode tmp=null;
	  while(next!=null){
		if(cur==list && cur!=null)
			cur.next=null;
            //save the third one
	    tmp = next.next;
            //reverse pointer
	    next.next = cur;
            //move cur and next
	    cur=next;
	    next=tmp;
	    
	  }

	  return cur;
	}
