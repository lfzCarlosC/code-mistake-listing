       	public ListNode reverse(ListNode list){
      if(list == null) return null;

	  ListNode next = list.next;
	  ListNode cur = list;
	  ListNode tmp=null;
	  while(next!=null){
		if(cur==list && cur!=null)
			cur.next=null;
            //save the third one
	    tmp = next.next;
            //reverse pointer
	    next.next = cur;
            //move cur and next
	    cur=next;
	    next=tmp;
	    
	  }

	  return cur;
	}

  ListNode deleteNode(ListNode head, int val){
    if( head==null ) return;
    ListNode dummy = new ListNode(-1);
    dummy.next=head;
    ListNode prev= dummy;
    ListNode cur= head;

    while(cur!=null){
      if(cur.val==val){
        prev.next = cur.next 
      }else{
        prev = cur;
      }
      cur = cur.next;

    }
    return dummy.next;   
}
